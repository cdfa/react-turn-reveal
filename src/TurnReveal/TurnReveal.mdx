# TurnReveal

A `<TurnReveal>` component is a component that can turn away or turn into view.

import { Playground, Props } from "docz";
import TurnReveal, { Direction, Transition } from "./TurnReveal";
import styled from "styled-components";
import Placeholder from "src/placeholder.png";

<Props of={TurnReveal} />

<Playground>
    {() => {
        // For now we have to do template literals like this. See https://github.com/mdx-js/mdx/issues/195
        const Front = styled.div(["\
            height: 100%;\
            width: 100%;\
            background: #1fb6ff;\
            display: flex;\
            justify-content: center;\
            flex-direction: column;\
            text-align: center;\
        "]);

        const Image = styled.img(["\
            width: 100%;\
            max-width: 400px;\
            height: auto;\
        "])

        return (
            <TurnReveal
                transition={Transition.show}
                direction={Direction.right}
                back={
                    <Image src={Placeholder} alt="Placeholder" width="40%"/>
                }>
                <Front>
                    More info
                </Front>
            </TurnReveal>
        )
    }}

</Playground>
